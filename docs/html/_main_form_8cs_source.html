405  private void interpolatedPictureBox1_Click(object sender, EventArgs e)
406  {
407  IGrid grid = game.Grid.Clone();
408  PictureBox pictureBox = sender as PictureBox;
409  MouseEventArgs mouseEventArgs = e as MouseEventArgs;
410  if (mouseEventArgs == null || pictureBox == null)
411  return;
412  int gridX = (int)(((double)mouseEventArgs.X / pictureBox.Width) * grid.XSize);
413  int gridY = (int)(((double)mouseEventArgs.Y / pictureBox.Height) * grid.YSize);
414  if (mouseEventArgs.Button == MouseButtons.Left)
415  grid[gridX, gridY] = true;
416  else if (mouseEventArgs.Button == MouseButtons.Right)
417  grid[gridX, gridY] = false;
418  game.Grid = grid;
